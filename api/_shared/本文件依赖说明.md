使用示例

例如：api/boxes/create-open-order.ts

import type { VercelRequest, VercelResponse } from '@vercel/node';
import { z } from 'zod';
import { withApiHandler, sendSuccess } from '../shared/response';
import { validateBody } from '../shared/validate';

const createOpenOrderSchema = z.object({
boxId: z.string().uuid(),
drawCount: z.union([z.literal(1), z.literal(10)]),
idempotencyKey: z.string().min(16).max(128),
});

export default async function handler(req: VercelRequest, res: VercelResponse) {
await withApiHandler(req, res, {
methods: ['POST'],
handler: async ({ req, res }) => {
const body = await validateBody(req, createOpenOrderSchema);

      // 这里继续：
      // 1. requireSession(req)
      // 2. 调用 Supabase RPC: gacha_create_order
      // 3. 返回 invoice payload

      sendSuccess(res, {
        boxId: body.boxId,
        drawCount: body.drawCount,
        invoicePayload: 'mock_invoice_payload',
      });
    },

});
}

需要安装依赖：

pnpm add zod
pnpm add -D @vercel/node

这三份文件默认依赖这些环境变量：

SUPABASE_URL
SUPABASE_SERVICE_ROLE_KEY
CORS_ALLOWED_ORIGINS
SESSION_COOKIE_NAME # 可选，默认 tma_game_session

对应数据库至少需要这些表：

core.app_sessions
core.users
